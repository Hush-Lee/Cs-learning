
	将所有节点左右节点互换，递归调用
```c++
	class Solution {

public:

TreeNode* mirrorTree(TreeNode* root) {

if(!root){

return root;

}

if(root->left||root->right){

TreeNode * tmp=root->left;

root->left=root->right;

root->right=tmp;

mirrorTree(root->left);

mirrorTree(root->right);

}

return root;

}

};
```
