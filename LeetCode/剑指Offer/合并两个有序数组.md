### 双指针乃至四指针都想过了，想不明白就换个方向
## 逆序双指针，第一个数组足以容纳两个数组，从后面填充不会发生覆盖
```cpp
class Solution {

public:

void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {

int now = m+n-1;

int i= m-1;

int j=n-1;

while(i>=0||j>=0){

if(j<0){

break;

}else if(i<0){

nums1[now--]=nums2[j--];

}else if(nums1[i]<nums2[j]){

nums1[now--]=nums2[j--];

} else{

nums1[now--]=nums1[i--];

}

}

}

};
```
