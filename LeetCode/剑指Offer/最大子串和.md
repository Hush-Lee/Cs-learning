	问题转化为以当前数字结尾的子串的最大值，以前一个数字结尾的最大值加上当前数，小则不要，大则保留，再使用一个变量，将最大值记录，返回即可
```cpp
class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        if(nums.size()==0){
            return 0;
        }
        int sum=0;
        int maxSum=nums[0];
        for (const auto &x: nums){
            sum+=x;
            sum=max(sum,x);
            maxSum=max(maxSum,sum);
        }
        return maxSum;
    }
};
```
