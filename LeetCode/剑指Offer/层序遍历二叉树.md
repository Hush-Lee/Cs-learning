- 层序遍历二叉树
- 将根节点放入队列，每次读取节点都将左右节点放入队列中（如果存在）
- 直到队列为空
```c++
class Solution {

public:

	vector<int> levelOrder(TreeNode* root) {
	queue <TreeNode*> treeQueue;
	treeQueue.push(root);
	vector<int> result;
	if(!root){
		return result;
	}
	while(!treeQueue.empty()){
		if(treeQueue.front()->left){
			treeQueue.push(treeQueue.front()->left);
		}
		if(treeQueue.front()->right){
			treeQueue.push(treeQueue.front()->right);
		}
		result.push_back(treeQueue.front()->val);
		treeQueue.pop();
	}
	return result;

}

};

```
- 每层隔行输出，将根节点后面放一个空节点；
- 每次遇到空节点都再次在最后放一个空节点；
- 连续两个空节点说明遍历完成
```c++
class Solution {

public:

vector<vector<int>> levelOrder(TreeNode* root) {

queue <TreeNode*> treeQueue;

treeQueue.push(root);

treeQueue.push(NULL);

vector<vector<int>> result;

vector<int> tmp;

if(!root){

return result;

}

while(!treeQueue.empty()){

if(treeQueue.front()==NULL){

treeQueue.push(NULL);

result.push_back(tmp);

treeQueue.pop();

if(treeQueue.front()==NULL){

break;

}

vector<int>().swap(tmp);

continue;

}

if(treeQueue.front()->left){

treeQueue.push(treeQueue.front()->left);

}

if(treeQueue.front()->right){

treeQueue.push(treeQueue.front()->right);

}

tmp.push_back(treeQueue.front()->val);

treeQueue.pop();

}

return result;

}

};
```

```c++
class Solution {

public:

vector<vector<int>> levelOrder(TreeNode* root) {

queue <TreeNode*> treeQueue;

treeQueue.push(root);

treeQueue.push(NULL);

int flag=1;

if(!root){

return result;

while(!treeQueue.empty()){

treeQueue.push(NULL);

if(flag==-1){

treeQueue.push(treeQueue.front()->left);

treeQueue.push(treeQueue.front()->right);
```
